% Section headers: first one (name) is centered, others left-aligned with underline
\newcommand{\doxysection}[1]{%
    \ifnum\value{doxymainsection}=0%
        % First section is the mainpage title (name) - centered, larger
        \par\addvspace{0.2em}%
        \begin{center}{\fontsize{16pt}{20pt}\selectfont\bfseries #1}\end{center}%
        \par\addvspace{0.2em}%
        \stepcounter{doxymainsection}%
    \else%
        % Subsequent sections - left-aligned, bold, with underline
        \par\addvspace{1.0em}{\Large\bfseries #1}\par\vspace{-0.2em}\rule{\textwidth}{0.5pt}\par\vspace{0.4em}%
    \fi%
}
