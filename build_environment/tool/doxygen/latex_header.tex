% Section headers: first one (name) is centered, others left-aligned with underline
\newcommand{\doxysection}[1]{%
    \ifnum\value{doxymainsection}=0%
        % First section is the mainpage title (name) - centered
        \par\addvspace{0.2em}%
        {\centering{\fontsize{22pt}{24pt}\selectfont\bfseries #1}\par}%
        \vspace{0.2em}%
        \stepcounter{doxymainsection}%
    \else%
        % Subsequent sections - centered, bold, with underline
        \par\addvspace{0.6em}\begin{center}{\Large\bfseries #1}\end{center}\par\vspace{-0.6em}\rule{\textwidth}{0.6pt}\par\vspace{0.4em}%
    \fi%
}
