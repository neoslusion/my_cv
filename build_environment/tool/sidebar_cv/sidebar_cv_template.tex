% Colors - Professional blue theme

\definecolor{accentcolor}{HTML}{3498DB}

    colorlinks=true,

    urlcolor=accentcolor,

% Section formatting for main content
\titleformat{\section}
{\normalfont\large\bfseries\color{textdark}}
{}{0em}{}
[\vspace{-0.3em}\textcolor{accentcolor}{\rule{\linewidth}{1pt}}\vspace{0.2em}]

\titlespacing*{\section}{0pt}{0.5em}{0.3em}

% Sidebar section command
\newcommand{\sidebarsection}[1]{%
    \vspace{0.5em}
    {\large\bfseries\textcolor{textlight}{#1}}\\[0.3em]
    \textcolor{accentcolor}{\rule{0.9\linewidth}{1pt}}
    \vspace{0.3em}
}

% Skill item with icon
\newcommand{\skillcategory}[2]{%
    \textcolor{accentcolor}{\small\faIcon{#1}} \textbf{\textcolor{textlight}{#2}}\\[0.2em]
}

% Skill tags
\newcommand{\skilltag}[1]{%
    \tikz[baseline=(char.base)]{
        \node[fill=accentcolor!20, text=textlight, rounded corners=2pt, inner sep=2pt, font=\scriptsize] (char) {#1};
    }%
}

% Contact item
\newcommand{\contactitem}[2]{%
    \textcolor{accentcolor}{\faIcon{#1}} \textcolor{textlight}{\small #2}\\[0.3em]
}

% Experience entry
\newcommand{\experienceentry}[4]{%
    \textbf{#1}\\
    {\small\textcolor{subtextcolor}{#2 | #3}}\\[0.2em]
    #4
    \vspace{0.4em}
}

% Education entry
\newcommand{\educationentry}[4]{%
    \textbf{#1}\\
    {\small\textcolor{subtextcolor}{#2 | #3}}\\
    {\small #4}
    \vspace{0.3em}
}

% Compact itemize
\setlist[itemize]{
